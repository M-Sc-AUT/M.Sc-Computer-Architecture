	-- import std_logic from the IEEE library
	library IEEE;
	use IEEE.std_logic_1164.all;
	
	-- this is the entity
	entity ANDGATE is
	port ( 
	I1 : in std_logic_vector(7 downto 0) := (others => '0');
	I2 : in std_logic;
	O  : out std_logic);
	end entity ANDGATE;
	
	-- this is the architecture
	architecture RTL of ANDGATE is
	begin
	O <= I1 and I2;
	end architecture RTL;
\end{lstlisting}

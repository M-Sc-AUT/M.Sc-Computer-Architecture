\section{سوال دوازدهم}

تفاوت‌های چارچوب \lr{ForCES} و \lr{OpenFlow} را شرح دهید.

\begin{qsolve}
	\begin{enumerate}
		\item \textbf{معماری و هدف:}
		\begin{itemize}
			\item \lr{:ForCES} هدف اصلی \lr{ForCES} جداسازی عناصر کنترلی (\lr{Control Elements}) و فورواردینگ (\lr{Forwarding Elements}) با استفاده از پروتکل‌های استاندارد برای ارتباط بین آن‌ها است. این چارچوب برای استفاده در انواع شبکه‌ها (از جمله شبکه‌های سنتی و \lr{SDN}) طراحی شده است.
			\item \lr{:OpenFlow} به‌طور خاص برای پیاده‌سازی شبکه‌های تعریف‌شده با نرم‌افزار (\lr{SDN}) طراحی شده و کنترل متمرکز را فراهم می‌کند. این چارچوب بیشتر روی قوانین و جداول جریان (\lr{Flow Tables}) متمرکز است.
		\end{itemize}
		
		\item \textbf{ارتباط بین کنترل و فورواردینگ:}
		\begin{itemize}
			\item \lr{:ForCES} از یک پروتکل استاندارد برای تبادل اطلاعات و دستورات بین عناصر کنترلی و فورواردینگ استفاده می‌کند. این پروتکل امکان ارتباط پویا و انعطاف‌پذیر را فراهم می‌سازد.
			\item \lr{:OpenFlow} از یک کنترل‌کننده متمرکز برای مدیریت مستقیم جداول جریان و ارتباط با سوئیچ‌ها استفاده می‌کند. این ارتباط معمولاً از طریق پروتکل \lr{OpenFlow} انجام می‌شود.
		\end{itemize}
		
		\item \textbf{انعطاف‌پذیری:}
		\begin{itemize}
			\item \lr{:ForCES} انعطاف‌پذیری بیشتری دارد و می‌تواند در محیط‌های سخت‌افزاری، نرم‌افزاری، یا ترکیبی استفاده شود. همچنین از معماری‌های مختلف شبکه پشتیبانی می‌کند.
			\item \lr{:OpenFlow} تمرکز بیشتری روی شبکه‌های \lr{SDN} دارد و انعطاف‌پذیری آن به محدوده قوانین تعریف‌شده توسط کنترل‌کننده محدود است.
		\end{itemize}
		
		\item \textbf{مدل داده:}
		\begin{itemize}
			\item \lr{:ForCES} از مدل داده‌ای استاندارد و قابل‌برنامه‌ریزی استفاده می‌کند که توسط \lr{IETF} تعریف شده است. این مدل داده برای توسعه‌دهندگان امکان توسعه آسان‌تر را فراهم می‌کند.
			\item \lr{OpenFlow:} مدل داده ساده‌تری دارد که بر اساس جداول جریان عمل می‌کند و برای شبکه‌های پویا طراحی شده است.
		\end{itemize}
		
		\item \textbf{محدوده کاربرد:}
		\begin{itemize}
			\item \lr{:ForCES} برای طیف گسترده‌ای از شبکه‌ها از جمله شبکه‌های سنتی، مجازی، و شبکه‌های \lr{SDN} مناسب است.
			\item \lr{OpenFlow:} بیشتر در شبکه‌های \lr{SDN} و برای کنترل متمرکز استفاده می‌شود.
		\end{itemize}
		
		\item \textbf{پشتیبانی از استانداردها:}
		\begin{itemize}
			\item \lr{:ForCES} یک استاندارد باز است که توسط \lr{IETF} توسعه داده شده و به‌عنوان یک چارچوب مستقل عمل می‌کند.
			\item \lr{:OpenFlow} توسط \lr{ONF (Open Networking Foundation)} توسعه داده شده و به‌طور خاص برای شبکه‌های \lr{SDN} طراحی شده است.
		\end{itemize}
	\end{enumerate}
\end{qsolve}
